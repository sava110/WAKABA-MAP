library.db

CREATE TABLE "user" (
	"id"	INTEGER,
	"name"	TEXT NOT NULL UNIQUE,
	"tell"	TEXT NOT NULL,
	"pass"	TEXT NOT NULL,
	PRIMARY KEY("id" AUTOINCREMENT)
)

CREATE TABLE "book" (
	"id"	INTEGER,
	"b_id"	INTEGER NOT NULL,
	"u_id"	INTEGER NOT NULL,
	PRIMARY KEY("id" AUTOINCREMENT)
)

CREATE TABLE "book_info" (
	"id"	INTEGER,
	"title"	TEXT NOT NULL,
	"writer"	TEXT,
	"image"	TEXT,
	PRIMARY KEY("id" AUTOINCREMENT)
)

CREATE TABLE "borrow" (
	"id"	INTEGER,
	"u_id"	INTEGER NOT NULL,
	"s_id"	INTEGER NOT NULL,
	"time"	TEXT,
	//"waiting"	INTEGER NOT NULL,
	"can_lent"	INTEGER NOT NULL,
	PRIMARY KEY("id" AUTOINCREMENT)
)

CREATE TABLE "tag" (
	"id"	INTEGER,
	"b_id"	INTEGER NOT NULL,
	"name"	TEXT NOT NULL,
	PRIMARY KEY("id" AUTOINCREMENT)
)

CREATE TABLE "action" (
	"u_id"	INTEGER NOT NULL,
	"b_id"	INTEGER NOT NULL,
	"good"	INTEGER,
	"keep"	INTEGER,
	PRIMARY KEY("u_id","b_id")
)
